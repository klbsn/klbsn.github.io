using System.Collections; using System.Collections.Generic; using UnityEngine; public class moveSeaweed : MonoBehaviour { [SerializeField]private float weedSpeed = 1f; [SerializeField]private float randomHeightOffset =1f; void Update() { if(randomHeightOffset == 0){ transform.position = new Vector3(15, , transform.position.z); } float randomHeightAmount = UnityEngine.Random.Range(-randomHeightOffset, randomHeightOffset); transform.position += Time.deltaTime * weedSpeed * Vector3.left; if(transform.position.x <= -10) transform.position = new Vector3(15, randomHeightAmount, transform.position.z); } }
